// Namespace y refs dentro del contenedor
const perfilNs = $.namespace('perfilNs');
const { nombre, label, btnGuardar, item, btnDestacar } = $.refs('#refs-ns-root');

// Defaults en el namespace
perfilNs.ensure({ nombre: 'Invitado', destacado: false, guardadoCount: 0 });

// Two-way y salida formateada
perfilNs.nombre(nombre).val();
perfilNs.nombre(label).map(n => `PerfilNs: ${n || 'Invitado'}`).text();

// Duplicados con toggleClass sobre el namespace
perfilNs.destacado(item).toggleClass('highlight');

// Acciones en el namespace
perfilNs.destacado(btnDestacar).on('click', () => perfilNs.set('destacado', v => !v), { prevent: true });
perfilNs.nombre(btnGuardar).on('click', () => perfilNs.set('guardadoCount', c => (c || 0) + 1), { prevent: true });
