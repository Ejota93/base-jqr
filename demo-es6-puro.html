// Namespace para la demo de computed
const comp = $.namespace('computedDemo');
comp.ensure({ nivel: 25, label: '', firstName: '', lastName: '', fullName: 'Invitado', heightCm: 170, weightKg: 70, bmiLabel: '' });

// Inputs y salidas
comp.nivel($('#comp-nivel')).val();
comp.label($('#comp-label')).text();
comp.firstName($('#comp-first')).val();
comp.lastName($('#comp-last')).val();
comp.fullName($('#comp-full-name')).text();
comp.heightCm($('#comp-height')).val();
comp.weightKg($('#comp-weight')).val();
comp.bmiLabel($('#comp-bmi')).text();

// Computados
$.computed(comp.k('label'), [comp.k('nivel')], (n) => `Nivel: ${Number(n || 0)}`);
$.computed(comp.k('fullName'), [comp.k('firstName'), comp.k('lastName')], (f, l) => {
  const full = `${(f || '').trim()} ${(l || '').trim()}`.trim();
  return full || 'Invitado';
});
$.computed(comp.k('bmiLabel'), [comp.k('heightCm'), comp.k('weightKg')], (hCm, wKg) => {
  const hM = Number(hCm || 0) / 100;
  const w = Number(wKg || 0);
  if (hM <= 0 || w <= 0) return 'BMI: N/A';
  const bmi = w / (hM * hM);
  const rounded = Math.round(bmi * 10) / 10;
  let cat = 'Normal';
  if (rounded < 18.5) cat = 'Bajo peso'; else if (rounded < 25) cat = 'Normal'; else if (rounded < 30) cat = 'Sobrepeso'; else cat = 'Obesidad';
  return `BMI: ${rounded} (${cat})`;
});

15. Emisi√≥n con $.dispatch()

Emite eventos personalizados y deja que otros m√≥dulos los escuchen (p. ej. el Contador).

Nota: la tarjeta 1 (Contador) escucha estos eventos y actualiza su estado. 
// Emisi√≥n global por defecto
$.dispatch('contador-inc', { step: 1 });
$.dispatch('contador-dec', { step: 1 });
$.dispatch('contador-reset');

// Forma expl√≠cita: especificando el target
$.dispatch(document, 'contador-inc', { step: 1 });
$.dispatch(document, 'contador-dec', { step: 1 });
$.dispatch(document, 'contador-reset');

// Escucha cross-m√≥dulo (ejecutado en el m√≥dulo del contador)
$.events(document, {
  'contador-inc': (e, $root, detail) => {
    const step = detail?.step || 1;
    $.state('contador', c => (c || 0) + step);
  },
  'contador-dec': (e, $root, detail) => {
    const step = detail?.step || 1;
    $.state('contador', c => (c || 0) - step);
  },
  'contador-reset': () => $.state('contador', 0)
});

16. Alias $.fn.reactiveList()

Usa el alias sem√°ntico para listas reactivas directamente desde el contenedor.

// Estado inicial
$.ensure({ 'aliasList.items': [{ id: 1, text: 'Manzana' }, { id: 2, text: 'Banana' }] });

// Renderizador de item
function renderItem(item) {
  return `<div class="task-item" data-id="${item.id}">
    <span>${item.text}</span>
    <button class="btn-eliminar" data-ref="del">‚ùå</button>
  </div>`;
}

// Uso del alias sobre la clave
$('#alias-list').reactiveList('aliasList.items', renderItem, {
  key: 'id',
  placeholder: '<p style="text-align:center;color:#718096;">Lista vac√≠a</p>'
});

// Acciones
$('#alias-add').on('click', (e) => {
  e.preventDefault();
  $.state('aliasList.items', (prev) => [...(prev || []), { id: Date.now(), text: `Item ${(prev||[]).length + 1}` }]);
});
$('#alias-shuffle').on('click', (e) => {
  e.preventDefault();
  const items = [...($.state('aliasList.items') || [])];
  for (let i = items.length - 1; i > 0; i--) { const j = Math.floor(Math.random() * (i + 1)); [items[i], items[j]] = [items[j], items[i]]; }
  $.state('aliasList.items', items);
});
$('#alias-clear').on('click', (e) => { e.preventDefault(); $.state('aliasList.items', []); });

// Delegaci√≥n para eliminar
$.events('#alias-list', 'click', '.btn-eliminar', (e) => {
  e.preventDefault();
  const $item = $(e.target).closest('.task-item');
  const id = Number($item.data('id'));
  $.state('aliasList.items', (prev) => (prev || []).filter(it => it.id !== id));
});
