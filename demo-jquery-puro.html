function agregarElementoAnimado() {
  const n = $.state('totalElementos') + 1;
  const html = '
‚ú® Elemento #' + n + '
';
  const actual = $('#contenedor-animado').html();
  $.state('elementosHtml', actual.includes('Los elementos aparecer√°n aqu√≠') ? html : actual + html);
  $.state('totalElementos', n);
}

function removerElementoAnimado() {
  const $els = $('#contenedor-animado').children('.reactive-element');
  if ($els.length > 0) {
    $els.last().slideUp(300, function() {
      $(this).remove();
      const total = $.state('totalElementos') - 1;
      $.state('totalElementos', total);
      if (total === 0) {
        $.state('elementosHtml', '
Los elementos aparecer√°n aqu√≠
');
      }
    });
  }
}
